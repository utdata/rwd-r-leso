Texas higher education

Look for K-12 and HI_ED

tx %>% 
  filter(
    str_detect(agency_name, "HI_ED")
  ) %>% 
  group_by(agency_name) %>% 
  summarise(
    cnt = sum(quantity),
    total_value = sum(quantity * acquisition_value)
  ) %>% 
  arrange(total_value %>% desc())
